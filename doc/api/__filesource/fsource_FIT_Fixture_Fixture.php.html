  278.     public function checkCell( $cell, $a )

  279.     {
    280. 

  280.         $text = $cell->text();
    281. 

  281.         
    282. 

  282.         if( $text == '' ) {
    283. 

  283.         
    284.

    286.

    288. 

  288.             }
    289.

    291. 

  291.                 $this->info( $cell, $a->toString( $a->get() ) );
    292. 

  292.             } 
    293. 

  293.             catch( Exception $e ) {
    294. 

  294.                 $this->info( $cell, 'error' );
    295. 

  295.             }
    296. 

  296.             return true;                     
    297. 

  297.         } 
    298. 

  298.         
    299. 

  299.         if( $a == null ) {
    300. 

  300.             $this->ignore( $cell );
    301. 

  301.             return true;
    302. 

  302.         }
    303. 

  303.         
    304. 

  304.         if( strncmp( $text, 'error', 5 ) == 0 ) {
    305. 

  305.             try {
    306. 

  306.                 $result = $a->invoke();
    307. 

  307.                 $this->wrong( $cell, $a->toString() );
    308. 

  308.             }
    309. 

  309.             catch( Exception $e ) {
    310. 

  310.                 $this->right( $cell );
    311. 

  311.             }         
    312. 

  312.             return true;            
    313. 

  313.         } 
    314. 

  314.         
    315. 

  315.         try {
    316. 

  316.             // the value of the attribute or the return value of the method
    317. 

  317.             $result = $a->get(); 
    318. 

  318.             
    319. 

  319.             if( $a->equals( $a->parse( $text ), $result ) ) {
    320. 

  320.                 $this->right( $cell );
    321. 

  321.             } 
    322. 

  322.             else {
    323. 

  323.                 $result = $this->fixBoolToString( $result );
    324. 

  324.                 $this->wrong( $cell, $a->toString( $result ) );
    325. 

  325.             }
    326. 

  326.         }
    327. 

  327.         catch( Exception $e ) {
    328. 

  328.             $this->exception($cell, $e);
    329. 

  329.         }
    330. 

  330.         
    331. 

  331.         return true;
    332. 

  332.     }
