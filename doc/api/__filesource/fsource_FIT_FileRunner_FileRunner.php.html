  127.     public function run( $in, $out ) {

  128.     
    129. 

  129.         // check input file

  130.         if( !file_exists( $in ) ) {

  132.         }

  133.         if( !is_readable( $in ) ) {

  134.             throw new PHPFIT_Exception_FileIO( 'Input file does not exist!', $in );

  135.         }
    136. 

  136.

  137.         // check output file
    138. 

  138.         if( file_exists( $out ) ) {
    139. 

  139.             if( !is_writable( $out ) ) {
    140. 

  140.                 throw new PHPFIT_Exception_FileIO( 'Output file is not writable (probably a problem of file permissions)', $in );
    141. 

  141.             }
    142. 

  142.         }
    143. 

  143.         else {
    144. 

  144.             if( !is_writable( dirname( $out ) ) ) {

  145.                 throw new PHPFIT_Exception_FileIO( 'Cannot create output file in given folder. (probably a problem of file permissions)', $in );

  146.             }

  147.         }

    164.
